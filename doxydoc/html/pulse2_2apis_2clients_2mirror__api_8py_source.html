37  def getMirror(self, machine):
38  if self.initialized_failed:
39  return []
40  machine = self.convertMachineIntoH(machine)
41  uuid = machine['uuid']
42  try:
43  entity_uuid = ComputerLocationManager().getMachinesLocations([uuid])[uuid]['uuid']
44  parent_entities = [entity_uuid] + ComputerLocationManager().getLocationParentPath(entity_uuid)
45  url = ''
46  entity_uuid = ''
47  db = ImagingDatabase()
48  for _uuid in parent_entities:
49  urlsearch = db.getEntityUrl(_uuid)
50  if urlsearch is not None:
51  entity_uuid = _uuid
52  url = urlsearch
53  break;
54  Entity_Name = ComputerLocationManager().getLocationName(entity_uuid)
55  machine ['Entity_Name']= Entity_Name
56  serverinfo = db.getImagingServerInfo(entity_uuid)
57  machine ['entity_uuid']= entity_uuid
58  machine ['server']= urlparse(url).hostname
59  machine ['servernane']=serverinfo.name
60  except:
61  logging.getLogger().error("Cannot get Entity for this machine UUID (%s)" % uuid)
62  d = self.callRemote("getMirror", machine)
63  d.addErrback(self.onError, "MirrorApi:getMirror", machine)
64  return d

71  def _updateStateFile(self, data):
72  """
73  Update the RPC state file content.
74  """
75  assert(type(data) == dict)
76  self.logger.debug('Updating RPC replay file: %s' % self.filename)
77  fobj = file(self.filename, 'w')
78  self.logger.debug('Dumping and writing data')

102  @type first: bool

110  def pop(self, count):
111  """
112  Get count items from the RPC replay file. The file will be updated !
113 
114  @rtype: tuple
115  @returns: a tuple with the RPC to replay, or None if the store is empty
116  """
117  items = []
118  try:
119  self.logger.debug('Taking at most %d items from the RPC replay file' % count)
120  data = self.get()
121  if data:
122  timestamps = data.keys()
123  timestamps.sort()
124  found = 0
125  i = 0
126  while found != count and i < len(timestamps):
127  tstamp = timestamps[i]
128  # Is this item still valid ?
129  if time.time() - tstamp < 7 * 24 * 3600:
130  if data[tstamp][0] == self.PENDING:
131  item = [tstamp] + data[tstamp][1:]
132  items.append(item)
133  data[tstamp][0] = self.INPROGRESS
134  found += 1
135  else:
136  # Discard entry
137  self.logger.debug('Removing old entry: %s'
138  % str(data[tstamp]))
139  del data[tstamp]
140  i += 1
141  self._updateStateFile(data)
142  self.logger.debug('Done, found %d items' % len(items))
143  else:
144  self.logger.debug('RPC replay file is empty')
145  except Exception, e:
146  self.logger.exception(e)
147  return items

185 class RPCReplay(Singleton):
186 
187  """
188  This singleton class tries periodically to re-play the RPCs that were not
189  sent to the MMC agent and saved into the RPCStore object.
190 
191  The RPCs replay is scheduled every self.timer seconds.
192  self.count RPCs are replayed according to their timestamp order, with an
193  interval of self.interval seconds between each RPC so that we don't
194  overload the RPC server (the MMC agent).
195  """
196 
197  def init(self):
198  """
199  Initialize object, and the RPCStore object.
200  """
201  self.logger = logging.getLogger('imaging')
202  self.logger.debug('Initializing imaging RPC replay manager')
203  self.timer = PackageServerConfig().imaging_api['rpc_loop_timer']
204  self.count = PackageServerConfig().imaging_api['rpc_count']
205  self.interval = PackageServerConfig().imaging_api['rpc_interval']
206  self._replaying = False
207  self._delayedCall = None
208  self.store = RPCStore()
209  self.store.init()
210  reactor.addSystemEventTrigger('before', 'shutdown', self.cancelLoop)

237  def onError(self, error, funcname, args, default_return = [], timestamp = None):
238  """
239  Error back to be called when a XML-RPC call fails.
240  The RPC is saved so that it can be replayed later.
241  """
242  # Mutable list default_return used as default argument to a method or function
243  self.logger.warn('%s %s has failed: %s' % (funcname, args, error))
244  self.logger.info('Storing RPC, it will be replayed later')
245  self.store.add(funcname, args, timestamp)

261  def _run(self):
262  """
263  Schedule RPCs replay
264  """
265  def _rpcReplayCb(result):
266  self.logger.debug('All scheduled RPCs have been replayed')
267  self.store.discardRPCs(self._success)
268  self._replaying = False
269 
270  if self._replaying:
271  self.logger.debug('RPC replay already in progress')
272  else:
273  self.logger.debug('Starting scheduling of RPCs to replay')
274  self._replaying = True
275  self._success = []
276  try:
277  self.store.debug()
278  items = self.store.pop(self.count)
279  deferreds = []
280  interval = 0
281  for item in items:
282  timestamp, func, args = item
283  interval += self.interval
284  d = task.deferLater(reactor, interval,
285  self._replayRPC, *item)
286  deferreds.append(d)
287  dl = defer.DeferredList(deferreds)
288  dl.addCallback(_rpcReplayCb)
289  except Exception, e:
290  self.logger.exception('Error during RPC replay: %s', e)
291  self.logger.debug('End scheduling of RPCs to replay')
292  self.startLoop()
293 
294  def _replayRPC(self, timestamp, func, args):
295  """
296  Replay one RPC. If it succeed, the RPC timestamp is appended to
297  self._success list, which means the RPC will be removed from the state
298  file.
299  """
300  client = self._getXMLRPCClient()
301  d = client.callRemote(func, *args)
302  d.addCallbacks(lambda x : self._success.append(timestamp),
303  self.onError, errbackArgs = (func, args, 0, timestamp))
304  return d

81 class LauncherCallingProvider(type):
82  """
83  Provides a optimal choice of launcher and the remote calls.
84 
85  The choice is primarily based on the "geographic" detection
86  (preferred_network parameter), but in the case of insufficients
87  resources is selected a launcher with the lowest load.
88 
89  This provider can be implemented as a metaclass.
90  """
91  # All launchers URLs identified by name of launcher
92  launchers = {}
93  # A launcher evaulated as the nearest
94  default_launcher = None
95  # No need the launcher detection
96  #single_mode = len(launchers) == 1
97  @property
98  def single_mode(self):
99  return len(self.launchers) == 1

120  def _call(self, launcher, method, *args):
121  """
122  Calls a remote method on selected launcher.
123 
124  @param launcher: name of launcher
125  @type launcher: str
126 
127  @param method: method name to call
128  @type method: str
129 
130  @param args: arguments of called method
131  @type args: list
132 
133  @return: result of remote method
134  @rtype: Deferred
135  """
136  if launcher :
137  logging.getLogger().debug("Calling on launcher: method %s(%s)" % (method, str(args)))
138  uri = self.launchers[launcher]
139  proxy = getProxy(uri)
140  d = proxy.callRemote(method, *args)
141  d.addErrback(self._call_error, launcher, method)
142  return d
143  else :
144  return fail(NoLauncherError)

301  def _extract_best_candidate(self, results):
302  """
303  Based on statistics, selects the best launcher.
304 
305  @param results: list of statistics
306  @type results: list
307 
308  @return: selected launcher
309  @rtype: str
310  """
311  final_launcher = None
312  best_score = 0
313  for success, result in results :
314  if success :
315  stats, launcher = result
316  score = stats.slots.slottotal - stats.slots.slotused
317  if score > best_score:
318  best_score = score
319  final_launcher = launcher

321  return final_launcher
322  else :
323  logging.getLogger().warn("No free slots on launchers, operation aborted")
324  return None

424 def same_call(method):
425  """
426  Decorates a method to be call having the same call name and arguments
427 
428  @param method: decorated method
429  @type method: callable
430 
431  @return: xmlrpc calling provider
432  @rtype: Deferred
433  """
434  #Example of use :
435  #
436  # declaration like this
437  #
438  # @same_call
439  # def method_name(self, id, client, command)
440  # pass
441  #
442  # is equaled to

444  # def method_name(self, id, client, command):
445  # return self.call_method("method_name", id, client, command)

453 class RemoteCallProxy :
454  """
455  Provides the remote calls to launchers.
456 
457  Each launcher method must be declared here and must return
458  the implemented providing method call_method which ensures
459  a optimal choice of launcher.

463  Instance of this class can be created on start of scheduler
464  and reused for all calls.

467  __metaclass__ = LauncherCallingProvider

168 def unique(s):
169  """
170  Return a list of the elements in s, but without duplicates.
171 
172  For example, unique([1,2,3,1,2,3]) is some permutation of [1,2,3],
173  unique("abcabc") some permutation of ["a", "b", "c"], and
174  unique(([1, 2], [2, 3], [1, 2])) some permutation of
175  [[2, 3], [1, 2]].
176 
177  For best speed, all sequence elements should be hashable. Then
178  unique() will usually work in linear time.
179 
180  If not possible, the sequence elements should enjoy a total
181  ordering, and if list(s).sort() doesn't raise TypeError it's
182  assumed that they do enjoy a total ordering. Then unique() will
183  usually work in O(N*log2(N)) time.
184 
185  If that's not possible either, the sequence elements must support
186  equality-testing. Then unique() will usually work in quadratic
187  time.
188  """
189 
190  n = len(s)
191  if n == 0:
192  return []
193 
194  # Try using a dict first, as that's the fastest and will usually
195  # work. If it doesn't work, it will usually fail quickly, so it
196  # usually doesn't cost much to *try* it. It requires that all the
197  # sequence elements be hashable, and support equality comparison.
198  u = {}
199  try:
200  for x in s:
201  u[x] = 1
202  except TypeError:
203  u = None # move on to the next method
204 
205  if u != None:
206  return u.keys()
207  del u
208 
209  # We can't hash all the elements. Second fastest is to sort,
210  # which brings the equal elements together; then duplicates are
211  # easy to weed out in a single pass.
212  # NOTE: Python's list.sort() was designed to be efficient in the
213  # presence of many duplicate elements. This isn't true of all
214  # sort functions in all languages or libraries, so this approach
215  # is more effective in Python than it may be elsewhere.
216  try:
217  t = list(s)
218  t.sort()
219  except TypeError:
220  t = None # move on to the next method
221 
222  if t != None:
223  assert n > 0
224  last = t[0]
225  lasti = i = 1
226  while i < n:
227  if t[i] != last:
228  t[lasti] = last = t[i]
229  lasti += 1
230  i += 1
231  return t[:lasti]
232  else:
233  del t
234 
235  # Brute force is all that's left.
236  u = []
237  for x in s:
238  if x not in u:
239  u.append(x)
240  return u

471 def splitComputerPath(path):
472  """
473  Split the computer path according to this scheme:
474  profile:/entity1/entity2/computerName
475 
476  @raise TypeError: if the computer path is not valid
477  @returns: returns a tuple with (profile, entities, hostname, domain)
478  @rtype: tuple
479  """
480  # Get profile
481  m = re.match("^([a-zA-Z0-9-]*):(.*)$", path)
482  if m:
483  profile = m.group(1)
484  tail = m.group(2)
485  else:
486  profile = ''
487  tail = path
488 
489  # Split entity path and computer FQDN
490  entities, fqdn = posixpath.split(tail)
491 
492  if entities and entities != '/':
493  if not entities.startswith('/'):
494  raise TypeError
495  # Check entities
496  for entity in entities.split('/'):
497  checkEntityName(entity)
498  else:
499  entities = ''
500 
501  if '.' in fqdn:
502  hostname, domain = fqdn.split('.', 1)
503  else:
504  hostname = fqdn
505  domain = ''
506 
507  if domain and not re.match('^([a-zA-Z][a-zA-Z0-9-]*[a-zA-Z0-9]\.){0,10}[a-zA-Z][a-zA-Z0-9-]*[a-zA-Z0-9]$', domain):
508  raise TypeError('Bad domain: %s' % domain)
509 
510  if not re.match('^([a-zA-Z0-9][a-zA-Z0-9-_]*[a-zA-Z0-9])$', hostname):
511  raise TypeError('Bad hostname: %s' % hostname)
512 
513 
514  return (profile, entities, hostname, domain)

696 class HasSufficientMemory :
697  """
698  Can be used as a decorator to avoid executing the functions with high costs.
699 
700  When usage of memory is less than mem_limit, this decorator returns
701  decorated function, otherwise neg_ret_value.
702 
703  Examples:
704  ---------
705  @HasSufficientMemory(80)
706  def called_function(arg1, arg2, ..)
707  ...
708  return True
709 
710  @HasSufficientMemory(60, "NOK")
711  def called_function(arg1, arg2, ..)
712  ...
713  return "OK"
714  """
715 
716  def __init__(self, mem_limit, neg_ret_value=False):
717  """
718  @param mem_limit: limit (in percent) of memory usage
719  @type mem_limit: int
720 
721  @param neg_ret_value: returned value when memory usage is exceeded
722  @type neg_ret_value: object
723  """
724 
725  self.mem_limit = mem_limit
726  self.neg_ret_value = neg_ret_value
727 
728  def __call__(self, fnc, *args, **kwargs):
729  """
730  @param fnc: decorated function
731  @type fnc: function type
732 
733  @returns: decorated function or neg_ret_value
734  """
735 
736  def wrapped (*args, **kwargs) :
737  if psutil.virtual_memory().percent < self.mem_limit :
738  return fnc(*args, **kwargs)
739  else :
740  logging.getLogger().warn("Not enough memory to run '%s'" % fnc.__name__)
741  return self.neg_ret_value
742 
743  return wrapped

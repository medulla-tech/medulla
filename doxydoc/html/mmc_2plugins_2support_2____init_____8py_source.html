101  def data_extract(self, response):
102  """
103  Custom corrections to display the license info.
104 
105  @param data: decoded data returned from license server
106  @type data: list
107 
108  @return: transformed data
109  @rtype: dict
110  """
111  # -------------------
112  # format of response:
113  # -------------------
114  # [{"name": "Support - Niveau 1/2",
115  # "country": "FR",
116  # "data": {"hours": "De 7h \u00e0 18h, les jours ouvr\u00e9s.",
117  # "phone": "+33 12 13 14 15 23",
118  # "email": "support-fr@mandriva.com",
119  # "links": [
120  # {
121  # "url": "http://hot-line.com",
122  # "text": "Hot-line Form",
123  # },
124  # {
125  # "url": "http://example.com",
126  # "text": "Acheter une license"
127  # },
128  # ]
129  # }
130  # },
131  # {"name": "Unterst√ºtzung - Ebene 1/2",
132  # "country": "DE",
133  # "data": {"hours": "Von 8 bis 18",
134  # "phone": "+31 452 144 254",
135  # "email": "support-de@mandriva.com"
136  # }
137  # }]
138  #
139  # -----------------------
140  # will be transformed to:
141  # -----------------------
142  # {"name": "Support - Niveau 1/2",
143  # "country": "FR",
144  # "hours": "De 7h \u00e0 18h, les jours ouvr\u00e9s.",
145  # "phone": "+33 12 13 14 15 23",
146  # "email": "support-fr@mandriva.com"
147  # }
148 
149  if not isinstance(response, list):
150  logging.getLogger().warning("No license data; widget will not be updated.")
151  return None
152 
153  data = None
154  for pack in response:
155  if "country" in pack:
156  if pack["country"] == config.country:
157  data = pack
158  break
159 
160  if not data:
161  logging.getLogger().warning("No license data for country '%s'; widget will not be updated." % config.country)
162  return None
163 
164  if not "data" in data:
165  logging.getLogger().warning("No license data")
166  return None
167 
168  for key in ["phone", "email", "hours", "raw_content"]:
169  if key in data["data"]:
170  data[key] = data["data"][key]
171 
172  if 'links' in data['data']:
173  data['links'] = data['data']['links']
174 
175  if "phone" in data:
176  data["phone_uri"] = "tel:%s" % data["phone"].replace(".", "")\
177  .replace("-", "").replace(" ", "")
178 
179  if "email" in data:
180  data["email_uri"] = "mailto:%s?subject=%s" % (data["email"], config.install_uuid)
181 
182  del data["data"]
183 
184  return self.normalize(data)

186  def normalize(self, data):
187  """
188  Converts all elements with unicode escape characters to unicode chars.
189 
190  Includes also nested lists containing another dicts.
191 
192  @param data: extracted response from the license server
193  @type data: dict
194 
195  @return: converted data
196  @rtype: dict
197  """
198  for key, value in data.items():
199  if isinstance(value, list):
200  lst = []
201  for sub_data in value:
202  lst.append(self.normalize(sub_data))
203  data[key] = lst
204  else:
205  data[key] = value
206  return data

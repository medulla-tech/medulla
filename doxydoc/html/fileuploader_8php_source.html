152  $uploadDirectory .= '/' . $random_dir . '/';
153  if (!is_writable($uploadDirectory)){
154  return array('error' => "Server error. Upload directory isn't writable.");

177  $these = implode(', ', $this->allowedExtensions);
178  return array('error' => 'File has an invalid extension, it should be one of '. $these . '.');
179  }

182

186  $filename .= rand(10, 99);

189

461  def __filter_on(self, query):
462  """
463  Use the glpi.ini conf parameter filter_on to filter machines on some parameters
464  The request is in OR not in AND, so be carefull with what you want
465  """
466  ret = self.__filter_on_filter(query)
467  if type(ret) == type(None):
468  return query
469  else:
470  return query.filter(ret)

472  def __filter_on_filter(self, query):
473  if self.config.filter_on != None:
474  a_filter_on = []
475  for filter_key, filter_values in self.config.filter_on.items():
476  if filter_key == 'state':
477  self.logger.debug('will filter %s in (%s)' % (filter_key, str(filter_values)))
478  a_filter_on.append(self.machine.c.states_id.in_(filter_values))
479  if filter_key == 'type':
480  self.logger.debug('will filter %s in (%s)' % (filter_key, str(filter_values)))
481  a_filter_on.append(self.machine.c.computertypes_id.in_(filter_values))
482  if filter_key == 'entity':
483  self.logger.debug('will filter %s in (%s)' % (filter_key, str(filter_values)))
484  a_filter_on.append(self.machine.c.entities_id.in_(filter_values))
485  if filter_key == 'autoupdatesystems_id':
486  self.logger.debug('will filter %s in (%s)' % (filter_key, str(filter_values)))
487  a_filter_on.append(self.machine.c.autoupdatesystems_id.in_(filter_values))
488  if not filter_key in ('state','type','entity','autoupdatesystems_id') :
489  self.logger.warn('dont know how to filter on %s' % (filter_key))
490  if len(a_filter_on) == 0:
491  return None
492  elif len(a_filter_on) == 1:
493  return a_filter_on[0]
494  else:
495  return and_(*a_filter_on)
496  return None

598  def _generatePassword(self, password, scheme = None):
599  """
600  Generate a string suitable for the LDAP userPassword field
601 
602  @param password: password to hash
603  @type password: str
604 
605  @param scheme: LDAP password scheme to use (crypt or ssha)
606  @type scheme: str
607 
608  @returns: string suitable for the LDAP userPassword field
609  @rtype: str
610  """
611  if not scheme:
612  scheme = self.config.passwordscheme
613  # If the passwd has been encoded in the XML-RPC stream, decode it
614  if isinstance(password, xmlrpclib.Binary):
615  password = str(password)
616  salt = self._getSalt(scheme)
617  if scheme == "crypt":
618  userpassword = "{crypt}" + crypt.crypt(password, salt)
619  else:
620  ctx = _digest(password)
621  ctx.update(salt)
622  userpassword = "{SSHA}" + base64.encodestring(ctx.digest() + salt)

775  def _applyUserDefault(self, entry, default):
776  """
777  Prepare the modification of a user entry with default values.
778 
779  @param entry: current user entry
780  @type entry: dict
781 
782  @param default: default values to apply
783  @type default: dict
784 
785  @return: new user entry
786  @rtype: dict
787  """
788  entry = copy.deepcopy(entry)
789  for attribute, value in default.items():
790  # Search if modifiers have been specified
791  s = re.search("^\[(.*)\]", value)
792  if s:
793  modifiers = s.groups()[0]
794  # Remove modifiers from the string
795  value = re.sub("^\[.*\]", "", value)
796  else: modifiers = ""
797  # Interpolate value
798  if "%" in value:
799  for a, v in entry.items():
800  if type(v) == list:
801  v = v[0]
802  if type(v) == str:
803  if "/" in modifiers: v = delete_diacritics(v)
804  if "_" in modifiers: v = v.lower()
805  if "|" in modifiers: v = v.upper()
806  value = value.replace("%" + a + "%", v)
807  if value == "DELETE":
808  for key in entry.keys():
809  if key.lower() == attribute:
810  del entry[key]
811  break
812  elif value.startswith("+"):
813  for key in entry.keys():
814  if key.lower() == attribute:
815  entry[key] = entry[key] + value[1:].split(",")
816  break
817  else:
818  found = False
819  for key in entry.keys():
820  if key.lower() == attribute:
821  entry[key] = value
822  found = True
823  break
824  if not found: entry[attribute] = value
825  return entry

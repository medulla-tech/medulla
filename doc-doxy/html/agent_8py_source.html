231  """
232  Very similar to deferToThread, but also handles function that results

236  logger.debug("Execution time: %f" % (time.time() - start))

238  def _cbSuccess(result, deferred, start):
239  _printExecutionTime(start)

243  _printExecutionTime(start)

246  def _putResult(deferred, f, session, args, kwargs):
247  logger.debug("Using thread #%s for %s" % (threading.currentThread().getName().split("-")[2], f.__name__))
248  # Attach current user session to the thread
249  threading.currentThread().session = session
250  start = time.time()
251  try:
252  result = f(*args, **kwargs)
253  except:
254  f = failure.Failure()
255  reactor.callFromThread(deferred.errback, f)
256  else:
257  if isinstance(result, defer.Deferred):
258  # If the result is a Deferred object, attach callback and
259  # errback (we are not allowed to result to a Deferred)
260  result.addCallback(_cbSuccess, deferred, start)
261  result.addErrback(_cbFailure, deferred, start)
262  else:
263  _printExecutionTime(start)
264  reactor.callFromThread(deferred.callback, result)
265 
266  function = args[0]
267  context = args[1]
268  args = args[2:]
269  d = defer.Deferred()
270  reactor.callInThread(_putResult, d, function, context, args, kwargs)
271  return d
